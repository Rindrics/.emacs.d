** Tangling
Every time I save this file, ~org-babel~ tangles the configuration file ~init.el~.
#+begin_src emacs-lisp
  (leaf *tangling
    :config
    (setq user-emacs-directory "~/GitHub/gtd/dotfiles/.emacs.d/")
    (defun tangle-init ()
      ;; Tangle code blocks if current buffer is init.org or mylibrary.org
      (when (member (buffer-file-name)
                    `(,(expand-file-name (concat user-emacs-directory "init.org"))))
        (let ((prog-mode-hook nil))
          (org-babel-tangle)
          (find-file-noselect "init.el")
          (with-current-buffer "init.el"
            (eval-buffer))
           (byte-compile-file "init.el")
          (kill-buffer "init.el"))))
    (add-hook 'after-save-hook 'tangle-init))
#+end_src
The setting is immediately reflected because ~(tangle-init)~ calls ~(eval-buffer)~ inside of it.

      (interactive)
      (progn
        (set-buffer (get-buffer-create "*scratch*"))
        (funcall initial-major-mode)
        (erase-buffer)
        (when (and initial-scratch-message (not inhibit-startup-message))

        (or arg
            (progn

              (switch-to-buffer "*scratch*")))

              ;; *scratch* バッファの内容を保存したら
